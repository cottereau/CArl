72 void Stitch_Meshes::set_grid_constraints(const libMesh::Mesh & mesh_A, const libMesh::Mesh & mesh_B, double vol_tol )
73 {
74  libMesh::MeshTools::BoundingBox bbox_A = libMesh::MeshTools::bounding_box(mesh_A);
75  libMesh::MeshTools::BoundingBox bbox_B = libMesh::MeshTools::bounding_box(mesh_B);
76 
77  // Just to be sure, test if the bboxes intersect!
78  homemade_assert_msg(bbox_A.intersect(bbox_B),"Meshes' bounding boxes do not intersect!\n");
79 
80  // Set the (future) intersection bbox corners
81  std::vector<double> eps_candidates(3,0);
82  for(unsigned int iii = 0; iii < 3; ++iii)
83  {
84  m_Grid_MinPoint(iii) = std::max(bbox_A.min()(iii),bbox_B.min()(iii));
85  m_Grid_MaxPoint(iii) = std::min(bbox_A.max()(iii),bbox_B.max()(iii));
86  eps_candidates[iii] = (m_Grid_MaxPoint(iii) - m_Grid_MinPoint(iii))/m_GridN_min;
87  }
88 
89  m_eps = *std::min_element(eps_candidates.begin(),eps_candidates.end());
90 
91  for(unsigned int iii = 0; iii < 3; ++iii)
92  {
93  m_Grid_MinPoint(iii) -= 2*m_eps;
94  m_Grid_MaxPoint(iii) += 2*m_eps;
95  }
96 
97  for(unsigned int iii = 0; iii < 3; ++iii)
98  {
99  m_GridN[iii] = (m_Grid_MaxPoint(iii) - m_Grid_MinPoint(iii)) / m_eps + 1;
100  }
101 
102  if( vol_tol < 0 )
103  {
104  // Grossily estimate the volume of A's and B's elements using the bbox
105  double grid_volume = (m_Grid_MaxPoint(0) - m_Grid_MinPoint(0)) *
106  (m_Grid_MaxPoint(1) - m_Grid_MinPoint(1)) *
107  (m_Grid_MaxPoint(2) - m_Grid_MinPoint(2));
108 
109 
110  double fraction_vol_A = (bbox_A.max()(0) - bbox_A.min()(0)) *
111  (bbox_A.max()(1) - bbox_A.min()(1)) *
112  (bbox_A.max()(2) - bbox_A.min()(2)) /
113  grid_volume;
114 
115  double fraction_vol_B = (bbox_B.max()(0) - bbox_B.min()(0)) *
116  (bbox_B.max()(1) - bbox_B.min()(1)) *
117  (bbox_B.max()(2) - bbox_B.min()(2)) /
118  grid_volume;
119 
120  unsigned int est_elem = std::max(fraction_vol_A * mesh_A.n_elem(),fraction_vol_B * mesh_B.n_elem());
121 
122  m_vol_tol = 1E-6 * grid_volume / est_elem;
123  }
124  else
125  {
126  m_vol_tol = vol_tol;
127  }
128 
129  // Mark grid as ready to use
130  m_bGridDefined = true;
131 
132  if(m_bPrintDebug)
133  {
134  std::cout << " DEBUG: discrete grid" << std::endl;
135  std::cout << " -> eps : " << m_eps << std::endl;
136  std::cout << " -> volume : " << m_vol_tol << std::endl;
137  std::cout << " -> Grid dimensions : " << m_GridN[0] << " " << m_GridN[1] << " " << m_GridN[2] << " " << std::endl << std::endl;
138  }

163 void Stitch_Meshes::join_tables()

179  unsigned int temp_nb_of_nodes = 0;
180

182  {

191  m_nb_of_elements += temp_nb_of_elements;
192  m_maximum_nb_of_nodes += temp_nb_of_nodes;
193  }
194 
195  // -> Preallocate the data structures
196  preallocate_grid(100*m_maximum_nb_of_nodes);
197 
198  m_Stitched_mesh.reserve_elem(m_nb_of_elements);
199  m_Stitched_mesh.reserve_nodes(m_maximum_nb_of_nodes);
200 
201  m_intersection_pairs.resize(m_nb_of_intersections);
202  m_intersection_nb_of_elements.resize(m_nb_of_intersections);
203  m_restriction_set_first.reserve(m_nb_of_intersections);
204  m_restriction_set_second.reserve(m_nb_of_intersections);
205 
206  // -> Second, read the data that will be used to reconstruct the
207  // intersection tables in the end: intersection pairs, and number of
208  // elements per intersection.
209  unsigned int intersection_idx = 0;
210  unsigned int dummy_uint = 0;
211  unsigned int nb_of_elems = 0;

214  table_file.open(m_table_filenames[iii]);
215  table_file >> temp_nb_of_intersections;
216  table_file >> temp_nb_of_elements;
217  table_file >> temp_nb_of_nodes;
218 
219  for(unsigned int jjj = 0; jjj < temp_nb_of_intersections; ++jjj)
220  {
221  table_file >> dummy_uint;
222  table_file >> m_intersection_pairs[intersection_idx].first
223  >> m_intersection_pairs[intersection_idx].second;
224  table_file >> m_intersection_nb_of_elements[intersection_idx];
225  carl::jump_lines(table_file);
226 
227  // Prepare data for the restriction mesh
228  nb_of_elems += m_intersection_nb_of_elements[intersection_idx];
229  m_restriction_set_first.insert(m_intersection_pairs[intersection_idx].first);
230  m_restriction_set_second.insert(m_intersection_pairs[intersection_idx].second);
231  ++intersection_idx;
232  }
233 
234  table_file.close();
235  }
236 
237  // -> Fourth, re-build the intersection tables
238  unsigned int intersection_elem_idx = 0;
239 
240  std::ofstream joined_tables_file(m_table_output);
241  joined_tables_file << m_nb_of_intersections << " "
242  << nb_of_elems << std::endl;
243 
244  for(unsigned int iii = 0; iii < m_nb_of_intersections; ++iii)
245  {
246  joined_tables_file << iii << " "
247  << m_intersection_pairs[iii].first << " "
248  << m_intersection_pairs[iii].second << " "
249  << m_intersection_nb_of_elements[iii] << " ";
250  for(unsigned jjj = 0; jjj < m_intersection_nb_of_elements[iii]; ++jjj)
251  {
252  joined_tables_file << intersection_elem_idx << " ";
253  ++intersection_elem_idx;
254  }
255  joined_tables_file << std::endl;
256  }
257  joined_tables_file.close();
258 }

260 void Stitch_Meshes::stitch_meshes()
261 {
262  homemade_assert_msg(m_bGridPreallocated,"Grid not preallocated!\n");
263 
264  // -> Third, stitch the meshes
265  libMesh::ReplicatedMesh temp_mesh(m_world_comm,3);
266  temp_mesh.allow_renumbering(false);
267 
268  unsigned int full_mesh_nb_elems = 0;
269  unsigned int full_mesh_nb_nodes = 0;
270 
271  libMesh::Elem * copy_elem = NULL;
272  libMesh::Elem * mesh_elem = NULL;
273  libMesh::Node * mesh_node = NULL;
274 
275  double dummy_volume = 0;
276 
277  // -> Stitch!
278  for(unsigned int iii = 0; iii < m_nb_files; ++iii)
279  {
280  // -> Open mesh file
281  temp_mesh.read(m_mesh_filenames[iii]);
282 
283  // -> Insert nodes
284  libMesh::ReplicatedMesh::element_iterator it_mesh = temp_mesh.elements_begin();
285 
286  for( ; it_mesh != temp_mesh.elements_end(); ++it_mesh)
287  {
288  copy_elem = * it_mesh;
289 
290  // -> Each element is unique, so no tests for insertion
291  mesh_elem = libMesh::Elem::build(libMesh::TET4).release();
292  mesh_elem->set_id(full_mesh_nb_elems);
293  mesh_elem->processor_id(0);
294 
295  // -> First, add the nodes
296  for(unsigned int jjj = 0; jjj < 4; ++jjj)
297  {
298  convert_to_discrete(copy_elem->point(jjj),m_dummy_discrete_point);
299  if(m_discrete_vertices.find(m_dummy_discrete_point) == m_discrete_vertices.end())
300  {
301  // New vertex! Add it to the mesh
302  m_discrete_vertices[m_dummy_discrete_point] = full_mesh_nb_nodes;
303  mesh_node = m_Stitched_mesh.add_point(copy_elem->point(jjj),full_mesh_nb_nodes,0);
304  ++full_mesh_nb_nodes;
305  }
306  else
307  {
308  mesh_node = m_Stitched_mesh.node_ptr(m_discrete_vertices[m_dummy_discrete_point]);
309  }
310 
311  // Associate vertex to the new element
312  mesh_elem->set_node(jjj) = mesh_node;
313 
314  }
315 
316  m_Stitched_mesh.add_elem(mesh_elem);
317  dummy_volume += mesh_elem->volume();
318 
319  ++full_mesh_nb_elems;
320  }
321  }
322 
323  // Print information about the number of collisions
324  if(m_bPrintDebug)
325  {
326  size_t collisions = 0;
327  for (size_t bucket = 0; bucket != m_discrete_vertices.bucket_count(); ++bucket)
328  {
329  if (m_discrete_vertices.bucket_size(bucket) > 1)
330  {
331  collisions += m_discrete_vertices.bucket_size(bucket) - 1;
332  }
333  }
334 
335  std::cout << " DEBUG: discrete grid hash collisions" << std::endl;
336  std::cout << " -> Nb. of collisions / size : " << collisions << " / " << m_discrete_vertices.size()
337  << " (" << 100.*collisions/m_discrete_vertices.size() << "%)" << std::endl << std::endl;
338  }
339  m_Stitched_mesh.prepare_for_use();
340 
341  // Print mesh
342  libMesh::NameBasedIO output_mesh(m_Stitched_mesh);
343  output_mesh.write(m_mesh_output);
344 
345  if(m_bPrintDebug)
346  {
347  std::cout << " DEBUG: stitched mesh" << std::endl;
348  std::cout << " -> Volume : " << dummy_volume << std::endl << std::endl;
349  }
350 
351  int wrong_volume = 0;
352  libMesh::ReplicatedMesh::element_iterator elem_begin = m_Stitched_mesh.local_elements_begin();
353  libMesh::ReplicatedMesh::element_iterator elem_end = m_Stitched_mesh.local_elements_end();
354 
355  for( ; elem_begin != elem_end; ++elem_begin)
356  {
357  libMesh::Elem * dummy_elem = * elem_begin;
358  if(std::abs(dummy_elem->volume()) < m_vol_tol)
359  {
360  ++wrong_volume;
361  }
362  }
363 
364  std::cout << " -> bad volumes : " << wrong_volume << " ( " << m_vol_tol << " ) " << std::endl;
