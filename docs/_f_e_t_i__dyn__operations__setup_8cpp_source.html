102  void FETI_Dyn_Operations::prepare_force_vector_by_modal_and_sinus(std::string force_path,
103  carl::DynSystemVectorPath* vectors,
104  double amplitude,
105  double frequency,
106  double initialPhase,
107  double small_deltat,
108  int index,
109  int timestep)
110  {
111  Vec modal;
112  Vec forceThis,forceNext;
113 
114  VecCreate(m_comm.get(),&modal);
115  carl::read_PETSC_vector(modal,force_path,m_comm.get());
116  VecDuplicate(modal,&forceThis);
117  VecDuplicate(modal,&forceNext);
118 
119 
120  VecCopy(modal,forceThis);
121  VecScale(forceThis,amplitude*sin(2*M_PI*frequency*small_deltat*index+initialPhase));
122  carl::write_PETSC_vector(forceThis,vectors->this_force,0,m_comm.get(),1);
123 
124  VecCopy(modal,forceNext);
125  VecScale(forceNext,amplitude*sin(2*M_PI*frequency*small_deltat*(index+timestep)+initialPhase));
126  carl::write_PETSC_vector(forceNext,vectors->next_force,0,m_comm.get(),1);
127 
128 
129  #ifdef PRINT_MATLAB_DEBUG
130  libMesh::PetscVector<libMesh::Number> vector_force_this(forceThis,m_comm);
131  vector_force_this.print_matlab(vectors->this_force+".m");
132  libMesh::PetscVector<libMesh::Number> vector_force_next(forceNext,m_comm);
133  vector_force_next.print_matlab(vectors->next_force+".m");
134  #endif
135 
136  VecDestroy(&modal);
137  VecDestroy(&forceThis);
138  VecDestroy(&forceNext);
