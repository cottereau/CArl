218  void assemble_coupling_matrices::set_corrected_shapes( const std::vector<std::vector<libMesh::Real> >& lambda_weights,
219  const std::vector<std::vector<libMesh::Real> >& phi_inter,
220  std::vector<std::vector<libMesh::Real> >& phi_corrected)
221  {
222  unsigned int lambda_corr_size = lambda_weights.size();
223  unsigned int lambda_inter_size = lambda_weights[0].size();
224 
225  unsigned int phi_inter_dof = phi_inter.size();
226  unsigned int phi_inter_qp = phi_inter[0].size();
227 
228  unsigned int phi_corrected_dof = phi_corrected.size();
229  unsigned int phi_corrected_qp = phi_corrected[0].size();
230 
231  /*
232  * lambda : [ n_dofs_corr ] x [ n_dofs_inter ]
233  * inter : [ n_dofs_inter ] x [ n_qp ]
234  * corr : [ n_dofs_corr ] x [ n_qp ]
235  */
236 
237  homemade_assert_msg(phi_inter_qp == phi_corrected_qp,
238  " Different numbers of quadrature points!");
239  homemade_assert_msg(lambda_corr_size == phi_corrected_dof,
240  " Incompatible corrected shape table and barycentric coordinates!");
241  homemade_assert_msg(lambda_inter_size == phi_inter_dof,
242  " Incompatible intersection shape table and barycentric coordinates!");
243 
244  // phi_A,i (qp) = l_i,1 * phi_I,1 (qp) + l_i,2 * phi_I,2 (qp) ...
245  // = sum [ l_i,j * phi_I,j (qp) ]
246  for(unsigned int qp = 0; qp < phi_inter_qp; ++qp)
247  {
248  for(unsigned int iii = 0; iii < phi_corrected_dof; ++iii)
249  {
250  phi_corrected[iii][qp] = 0;
251  for(unsigned int jjj = 0; jjj < phi_inter_dof; ++jjj)
252  {
253  phi_corrected[iii][qp] += lambda_weights[iii][jjj]*phi_inter[jjj][qp];
254  }
255  }
256  }
257  }

300  void assemble_coupling_matrices::get_lambdas( const unsigned int dim,
301  const libMesh::FEType& fe_t,
302  const libMesh::Elem* base_elem,
303  const std::vector<libMesh::Point>& phys_points,
304  std::vector<libMesh::Point>& ref_points,
305  std::vector<std::vector<libMesh::Real> >& lambda_weights)
306  {
307  // Test if we are using the correct type!
308  homemade_assert_msg(base_elem->type() == libMesh::TET4 || base_elem->type() == libMesh::HEX8,
309  " Only implemented for TET4 or HEX8 elements!");
310 
311  homemade_assert_msg(dim == 3,
312  " Only implemented for 3D!");
313 
314  unsigned int lambda_inter_size = lambda_weights[0].size();
315  unsigned int lambda_base_size = lambda_weights.size();
316 
317  // Set vectors to have same sizes
318  if(phys_points.size() != ref_points.size())
319  {
320  ref_points.resize( phys_points.size() );
321  }
322 
323  // Convert the points
324  libMesh::FEInterface::inverse_map(dim, fe_t, base_elem, phys_points, ref_points);
325 
326 
327 
328  // Calculate the lambdas
329  // -> lines : DoF from base
330  // -> cols : DoF from inter
331  for(unsigned int iii = 0; iii < lambda_base_size; ++iii)
332  {
333  for(unsigned int jjj = 0; jjj < lambda_inter_size; ++jjj)
334  {
335  lambda_weights[iii][jjj] =
336  libMesh::FE<3,libMesh::LAGRANGE>::shape(
337  base_elem->type(),
338  libMesh::FIRST,
339  iii,
340  ref_points[jjj]);
341  }
342  }
343  };
