29  const int col_N = col_addresses.dof_map.n_dofs();

33  // Vectors that will be used to calculate the preallocation. There are two
34  // "versions": a local one, starting at zero and which will contain this
35  // processor's preallocation after syncing, and a global one, which will be
36  // send to other processors
37  std::vector<unsigned int> local_n_nz(row_M_local,0);
38  std::vector<unsigned int> local_n_oz(row_M_local,0);

42  std::vector<int> end_row_DoF(nodes);
43  for(int iii = 0; iii < nodes; ++iii)
44  {

47  }
48

50  std::vector<int> end_col_DoF(nodes);
51  for(int iii = 0; iii < nodes; ++iii)
52  {

54  end_col_DoF[iii] = col_addresses.dof_map.end_dof(iii);
55  }

58  int local_end_row_DoF = end_row_DoF[rank];

60  // Iterate over the row's mesh cells
61 
62  int row_mesh_elem_idx = 0;
63  int dummy_col_dof;

66

71  row_addresses.mesh.active_local_elements_end();

73  // What the code has to do:
74  /*
75  * - For each element associated to the rows, get the intersecting
76  * elements and find out which processors own each of its DoFs. Use
77  * this to populate the "to_add_per_proc" table.
78  *
79  * - Once it is filled, sync the "to_add_per_proc" between the
80  * processors.
81  *
82  * - On each processor, iterate over the local row DoFs and add the
83  * info from "to_add_per_proc" to either "local_n_nz" or "local_n_oz",
84  * depending on the processor.
85  *
86  * - ... pray that it works?
87  */
88 
89  std::unordered_set<int> col_DoFs_to_allocate;
90  col_DoFs_to_allocate.reserve(col_addresses.mesh.n_nodes());
91 
92  for( ; it_row_elem != it_row_elem_end; ++it_row_elem )
93  {
94  // For each local mediator mesh elements ...
95  const libMesh::Elem* elem_mediator = *it_row_elem;
96 
97  // ... extract the element's index ...
98  row_mesh_elem_idx = elem_mediator->id();
99 
100  // ... set its DoFs ...
101  row_addresses.set_DoFs(row_mesh_elem_idx);
102 
103  // ... clear the col_DoFs set
104  col_DoFs_to_allocate.clear();
105 
106  // ... get the DoFs from the col mesh ...
107  auto range_elem_col_idx = inter_table.equal_range(row_mesh_elem_idx);
108 
109  for(auto it = range_elem_col_idx.first; it != range_elem_col_idx.second; ++it)
110  {
111  col_addresses.set_DoFs(it->second);
112 
113  for(unsigned int iii = 0; iii < col_addresses.n_dofs; ++iii)
114  {
115  col_DoFs_to_allocate.insert(col_addresses.dof_indices[iii]);
116  }
117  }
118 
119  // ... and finally increase the preallocation
120  for(auto it = col_DoFs_to_allocate.begin(); it != col_DoFs_to_allocate.end(); ++it)
121  {
122  dummy_col_dof = *it;
123 
124  for(int nnn = 0; nnn < nodes; ++nnn)
125  {
126  if( dummy_col_dof > begin_col_DoF[nnn] &&
127  dummy_col_dof < end_col_DoF[nnn])
128  {
129  for(unsigned int jjj = 0; jjj < row_addresses.dof_indices.size(); ++jjj)
130  {
131  ++to_add_per_proc[row_addresses.dof_indices[jjj]][nnn];
132  }
133  break;
134  }
135  }
136  }
137  }
138 
139  // Guarantee that everyone is in the same page
140  WorldComm.barrier();
141 
142  // Do a (lot) of syncs
143  for(int nnn = 0; nnn < nodes; ++nnn)
144  {
145  for(int iii = begin_row_DoF[nnn]; iii < end_row_DoF[nnn]; ++iii)
146  {
147  MPI_reduce_vector(to_add_per_proc[iii],nnn,WorldComm);
148  }
149  }
150 
151  int dummy_total_nz = 0;
152  int dummy_total_oz = 0;
153 
154  // Now, calculate the preallocation
155  for(int iii = local_begin_row_DoF; iii < local_end_row_DoF; ++iii)
156  {
157  for(int nnn = 0; nnn < rank; ++nnn)
158  {
159  local_n_oz[iii - local_begin_row_DoF] += to_add_per_proc[iii][nnn];
160  dummy_total_oz += to_add_per_proc[iii][nnn];
161  }
162 
163  local_n_nz[iii - local_begin_row_DoF] += to_add_per_proc[iii][rank];
164  dummy_total_nz += to_add_per_proc[iii][rank];
165 
166  for(int nnn = rank + 1; nnn < nodes; ++nnn)
167  {
168  local_n_oz[iii - local_begin_row_DoF] += to_add_per_proc[iii][nnn];
169  dummy_total_oz += to_add_per_proc[iii][nnn];
170  }
171  }
172 
173  dummy_total_nz = 0;
174  dummy_total_oz = 0;
175 
176  for(int iii = local_begin_row_DoF; iii < local_end_row_DoF; ++iii)
177  {
178  //local_n_nz[iii - local_begin_row_DoF] = 1.1*local_n_nz[iii - local_begin_row_DoF];
179  local_n_nz[iii - local_begin_row_DoF] = 2.0*local_n_nz[iii - local_begin_row_DoF];
180  local_n_nz[iii - local_begin_row_DoF] = std::max(static_cast<unsigned int>(30),local_n_nz[iii - local_begin_row_DoF]);
181  local_n_nz[iii - local_begin_row_DoF] = std::min(static_cast<unsigned int>(col_N_local),local_n_nz[iii - local_begin_row_DoF]);
182 
183  local_n_oz[iii - local_begin_row_DoF] = 1.1*local_n_oz[iii - local_begin_row_DoF];
184  local_n_oz[iii - local_begin_row_DoF] = std::max(static_cast<unsigned int>(10),local_n_oz[iii - local_begin_row_DoF]);
185  local_n_oz[iii - local_begin_row_DoF] = std::min(static_cast<unsigned int>(col_N - col_N_local),local_n_oz[iii - local_begin_row_DoF]);
186 
187  dummy_total_nz += local_n_nz[iii - local_begin_row_DoF];
188  dummy_total_oz += local_n_oz[iii - local_begin_row_DoF];
189  }
190 
191  coupling_matrix.init(row_M, col_N, row_M_local, col_N_local, local_n_nz, local_n_oz);
192 }
