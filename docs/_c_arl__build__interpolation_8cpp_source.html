260 void new_invert_PETSC_matrix(libMesh::Parallel::Communicator& WorldComm,
261  Mat& dense_matrix,
262  Mat& inv_matrix,
263  std::string inv_path)
264 {
265  Mat Ones,inv_dense_matrix;
266  PetscErrorCode ierr=0;
267  MatFactorInfo factor_info;
268  IS rperm, cperm;
269 
270  //Create identity and inverse of the dense matrix
271  MatDuplicate(dense_matrix,MAT_DO_NOT_COPY_VALUES,&inv_dense_matrix);
272  MatDuplicate(dense_matrix,MAT_DO_NOT_COPY_VALUES,&Ones);
273  MatSetType(inv_dense_matrix, MATSEQDENSE);
274  MatSeqDenseSetPreallocation(inv_dense_matrix,NULL);
275  MatSetType(Ones, MATSEQDENSE);
276  MatSeqDenseSetPreallocation(Ones,NULL);
277  MatZeroEntries(Ones);
278  MatShift(Ones,1);
279 
280  // Factor input matrix
281  MatGetOrdering(dense_matrix,MATORDERINGNATURAL, &rperm, &cperm);
282  MatFactorInfoInitialize(&factor_info);
283  MatLUFactor(dense_matrix,rperm,cperm,&factor_info);
284 
285  // Calculate inverse of the dense matrix
286  ierr = MatMatSolve(dense_matrix,Ones,inv_dense_matrix);
287 
288  // Reset input's factoring
289  MatSetUnfactored(dense_matrix);
290 
291  //Output inverse of the dense matrix
292  carl::write_PETSC_matrix(inv_dense_matrix,inv_path+".petscmat",0,PETSC_COMM_SELF,1);
293 
294  #ifdef PRINT_MATLAB_DEBUG
295  carl::write_PETSC_matrix_MATLAB(inv_dense_matrix,inv_path+".m",PETSC_COMM_SELF);
296  #endif
297 
298  // Cleanup
299  MatDestroy(&Ones);
300  MatDestroy(&dense_matrix);
301  MatDestroy(&inv_dense_matrix);
302 
303  //Load Matrix in WorldComm
304  MatCreate(WorldComm.get(),&inv_matrix);
305  carl::read_PETSC_matrix(inv_matrix, inv_path+".petscmat", WorldComm.get());
