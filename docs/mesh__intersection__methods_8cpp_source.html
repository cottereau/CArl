15 void Mesh_Intersection::triangulate_intersection(const std::set<libMesh::Point> & input_points)
16 {
17  m_libMesh_PolyhedronMesh.clear();
18 
19  for(std::set<libMesh::Point>::const_iterator it_set = input_points.begin();
20  it_set != input_points.end();
21  ++it_set)
22  {
23  libMesh::Point dummy = *it_set;
24  m_libMesh_PolyhedronMesh.add_point(*it_set);
25  }
26 
27  switch (m_MeshingMethod)
28  {
29  case IntersectionMeshingMethod::LIBMESH_TETGEN:
30  {
31  libMesh::TetGenMeshInterface temp_tetgen_interface(m_libMesh_PolyhedronMesh);
32  temp_tetgen_interface.triangulate_pointset();
33  break;
34  }
35  case IntersectionMeshingMethod::CGAL:
36  {
37  // Clear CGAL's mesh
38  m_CGAL_PolyhedronMesh.clear();
39 
40  // This is a superfluous extra point conversion:
41  // (CGAL Exact -> CGAL Inexact -> LibMesh -> CGAL Inexact)
42  // But changing this would need a re-structuring of the code (TODO !)
43  Point_3 dummy_CGAL_point;
44  const libMesh::Point * dummy_libMesh_point;
45  for(int iii = 0; iii < input_points.size(); ++iii)
46  {
47  dummy_libMesh_point = &m_libMesh_PolyhedronMesh.point(iii);
48  dummy_CGAL_point = Point_3( (*dummy_libMesh_point)(0),(*dummy_libMesh_point)(1),(*dummy_libMesh_point)(2));
49 
50  // Insert vertex and save the corresponding
51  Vertex_handle_3 dummy_vertex_handle = m_CGAL_PolyhedronMesh.insert(dummy_CGAL_point);
52  dummy_vertex_handle->info().ExtIndex = iii;
53  }
54 
55  // At this point, CGAL generated the mesh, now we have to convert it to libMesh
56  unsigned int dummy_vertex_idx = 0;
57  for( Finite_cells_iterator_3 cell_it = m_CGAL_PolyhedronMesh.finite_cells_begin();
58  cell_it != m_CGAL_PolyhedronMesh.finite_cells_end();
59  ++cell_it)
60  {
61  libMesh::Elem * elem_pointer = m_libMesh_PolyhedronMesh.add_elem(new libMesh::Tet4);
62  for(int iii = 0; iii < 4; ++iii)
63  {
64  dummy_vertex_idx = cell_it->vertex(iii)->info().ExtIndex;
65  elem_pointer->set_node(iii) = &(m_libMesh_PolyhedronMesh.node_ref(dummy_vertex_idx));
66  }
67  }
68  m_libMesh_PolyhedronMesh.prepare_for_use();
69 
70  break;
71  }
72  }

75 void Mesh_Intersection::update_intersection_mesh()
76 {
77  // Test which elements must be added, and insert their vertices
78  libMesh::ReplicatedMesh::element_iterator it_poly_mesh = m_libMesh_PolyhedronMesh.elements_begin();
79 
80  for( ; it_poly_mesh != m_libMesh_PolyhedronMesh.elements_end();
81  ++it_poly_mesh)
82  {
83  libMesh::Elem * poly_elem = * it_poly_mesh;
84 
85  if(std::abs(poly_elem->volume()) > m_vol_tol)
86  {
87  // Add a new element!
88  libMesh::Elem * mesh_elem = m_libMesh_Mesh.add_elem(new libMesh::Tet4);
89 
90  // -> First, add the nodes
91  for(unsigned int iii = 0; iii < 4; ++iii)
92  {
93  convert_to_discrete(poly_elem->point(iii),m_dummy_discrete_point);
94 
95  if(m_discrete_vertices.find(m_dummy_discrete_point) == m_discrete_vertices.end())
96  {
97  // New vertex! Add it to the mesh
98 
99  m_discrete_vertices[m_dummy_discrete_point] = m_nb_of_vertices;
100  m_libMesh_Mesh.add_point(poly_elem->point(iii),m_nb_of_vertices);
101  mesh_elem->set_node(iii) = m_libMesh_Mesh.node_ptr(m_nb_of_vertices);
102  ++m_nb_of_vertices;
103  }
104 
105  // Associate vertex to the new element
106  mesh_elem->set_node(iii) = m_libMesh_Mesh.node_ptr(m_discrete_vertices[m_dummy_discrete_point]);
107  }
108  // Increase the number of elements
109  ++m_nb_of_elements;
110  }
111  }

179 void Mesh_Intersection::set_grid_constraints(const libMesh::Mesh & mesh_A, const libMesh::Mesh & mesh_B, double vol_tol)
180 {
181  libMesh::BoundingBox bbox_A = libMesh::MeshTools::create_bounding_box(mesh_A);
182  libMesh::BoundingBox bbox_B = libMesh::MeshTools::create_bounding_box(mesh_B);
183 
184  // Just to be sure, test if the bboxes intersect!
185  homemade_assert_msg(bbox_A.intersects(bbox_B),"Meshes' bounding boxes do not intersect!\n");
186 
187  // Set the (future) intersection bbox corners
188  std::vector<double> eps_candidates(3,0);
189  for(unsigned int iii = 0; iii < 3; ++iii)
190  {
191  m_Grid_MinPoint(iii) = std::max(bbox_A.min()(iii),bbox_B.min()(iii));
192  m_Grid_MaxPoint(iii) = std::min(bbox_A.max()(iii),bbox_B.max()(iii));
193  eps_candidates[iii] = (m_Grid_MaxPoint(iii) - m_Grid_MinPoint(iii))/m_GridN_min;
194  }
195 
196  m_eps = *std::min_element(eps_candidates.begin(),eps_candidates.end());
197 
198  for(unsigned int iii = 0; iii < 3; ++iii)
199  {
200  m_Grid_MinPoint(iii) -= 2*m_eps;
201  m_Grid_MaxPoint(iii) += 2*m_eps;
202  }
203 
204  if( vol_tol < 0 )
205  {
206  // Grossily estimate the volume of A's and B's elements using the bbox
207  double grid_volume = (m_Grid_MaxPoint(0) - m_Grid_MinPoint(0)) *
208  (m_Grid_MaxPoint(1) - m_Grid_MinPoint(1)) *
209  (m_Grid_MaxPoint(2) - m_Grid_MinPoint(2));
210 
211 
212  double fraction_vol_A = (bbox_A.max()(0) - bbox_A.min()(0)) *
213  (bbox_A.max()(1) - bbox_A.min()(1)) *
214  (bbox_A.max()(2) - bbox_A.min()(2)) /
215  grid_volume;
216 
217  double fraction_vol_B = (bbox_B.max()(0) - bbox_B.min()(0)) *
218  (bbox_B.max()(1) - bbox_B.min()(1)) *
219  (bbox_B.max()(2) - bbox_B.min()(2)) /
220  grid_volume;
221 
222  unsigned int est_elem = std::max(fraction_vol_A * mesh_A.n_elem(),fraction_vol_B * mesh_B.n_elem());
223 
224  m_vol_tol = 1E-6 * grid_volume / est_elem;
225  }
226  else
227  {
228  m_vol_tol = vol_tol;
229  }
230 
231  for(unsigned int iii = 0; iii < 3; ++iii)
232  {
233  m_GridN[iii] = (m_Grid_MaxPoint(iii) - m_Grid_MinPoint(iii)) / m_eps + 1;
234  }
235 
236  if(m_bPrintDebug)
237  {
238  std::cout << " DEBUG: discrete grid" << std::endl;
239  std::cout << " -> eps : " << m_eps << std::endl;
240  std::cout << " -> volume : " << m_vol_tol << std::endl;
241  std::cout << " -> Grid dimensions : " << m_GridN[0] << " " << m_GridN[1] << " " << m_GridN[2] << " " << std::endl << std::endl;
242  }
243 }

245 void Mesh_Intersection::increase_intersection_mesh( const std::set<libMesh::Point> & input_points,
246  unsigned int elem_idx_A,
247  unsigned int elem_idx_B,
248  unsigned int elem_idx_C)
249 {
250  m_bMeshFinalized = false;
251  unsigned int intersection_range_start = m_nb_of_elements;
252 
253  // First, triangulate the point set!
254  triangulate_intersection(input_points);
255 
256  // Second, add the points to the grid-to-mesh map and update the intersection mesh,
257  // taking into account if the associated elements are big enough.
258  update_intersection_mesh();
259 
260  // Third, update the intersection pairs, if needed
261  unsigned int intersection_range_end = m_nb_of_elements;
262  if(intersection_range_end != intersection_range_start)
263  {
264  update_intersection_pairs(elem_idx_A, elem_idx_B,m_nb_of_intersections);
265  m_intersection_couplings[m_nb_of_intersections] = elem_idx_C;
266  update_intersection_element_range(intersection_range_start,intersection_range_end,m_nb_of_intersections);
267  ++m_nb_of_intersections;
268  }
