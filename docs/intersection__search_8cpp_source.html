119  /*
120  * Find the first intersecting pair from a patch. Do so by :
121  *
122  * a) locating the elements from the guide patch that contain the
123  * vertices of an element from the probed patch.
124  * b) for each one of these guide elements, find the one that
125  * intersects the probed element inside the coupling.
126  * c) if this fails, use a brute force search method.
127  */
128  void Intersection_Search::FindFirstPair( Patch_construction * Patch_guide,
129  Patch_construction * Patch_probed,
130  std::pair<unsigned int,unsigned int> & First_intersection)
131  {
132  // Set up some references for a simpler code
133  libMesh::ReplicatedMesh& Mesh_patch_guide = Patch_guide->patch_mesh();
134  libMesh::ReplicatedMesh& Mesh_patch_probed = Patch_probed->patch_mesh();
135 
136  // Set up locator for the first intersecting pair
137  std::unique_ptr<libMesh::PointLocatorBase> Patch_guide_Locator = Mesh_patch_guide.sub_point_locator();
138 
139  // Instruction needed to avoid the code from crashing if a query is
140  // outside the mesh
141  Patch_guide_Locator->enable_out_of_mesh_mode();
142 
143  // Find the first pair
144  libMesh::Elem * Patch_probed_first_elem = NULL;
145  unsigned int Patch_probed_first_elem_id = 0;
146 
147  // Set of intersecting terms:
148  std::set<unsigned int> Intersecting_guide_elems;
149  bool bFoundIntersection = false;
150  for(libMesh::ReplicatedMesh::element_iterator element_probed_it = Mesh_patch_probed.elements_begin();
151  element_probed_it != Mesh_patch_probed.elements_end();
152  ++element_probed_it)
153  {
154  Patch_probed_first_elem = * element_probed_it;
155  Patch_probed_first_elem_id = Patch_probed->convert_patch_to_parent_elem_id(Patch_probed_first_elem->id());
156  bFoundIntersection = m_Intersection_test.FindAllIntersection(Patch_probed_first_elem,Patch_guide_Locator,Intersecting_guide_elems);
157  if(bFoundIntersection)
158  {
159  break;
160  }
161  }
162 
163  // Search for one intersecting term inside the guide patch
164  libMesh::Elem * Patch_guide_first_elem = NULL;
165  unsigned int Patch_guide_first_elem_id;
166  bool bFoundFirstInter = false;
167 
168  for(std::set<unsigned int>::iterator it_inter = Intersecting_guide_elems.begin();
169  it_inter != Intersecting_guide_elems.end();
170  ++it_inter)
171  {
172  Patch_guide_first_elem = Mesh_patch_guide.elem_ptr(*it_inter);
173  Patch_guide_first_elem_id = Patch_guide->convert_patch_to_parent_elem_id(*it_inter);
174 
175  // Must test if the intersection is inside the coupling region
176  bFoundFirstInter = m_Intersection_test_neighbors.libMesh_exact_do_intersect_inside_coupling(Patch_probed_first_elem,Patch_guide_first_elem);
177 
178  if(bFoundFirstInter)
179  {
180  First_intersection.first = Patch_guide_first_elem_id;
181  First_intersection.second = Patch_probed_first_elem_id;
182  break;
183  }
184  }
185 
186  if(!bFoundFirstInter)
187  {
188  // Couldn't find the first intersection using a point search ...
189  // Will have to do the things the hard way ...
190  BruteForce_FindFirstPair(Patch_guide,Patch_probed,First_intersection);
191  }
192  };

239  /*
240  * Find all the intersections between the patches, using an advancing
241  * front method.
242  */
243  void Intersection_Search::FindPatchIntersections_Front(const libMesh::Elem * Query_elem)
244  {
245  m_perf_log.push("Preamble","Advancing front algorithm");
246  // Code for the front intersection tests
247  m_Intersection_Pairs_multimap.clear();
248 
249  // Intersection_Tools
250  m_Intersection_test.libmesh_set_coupling_nef_polyhedron(Query_elem);
251  m_Intersection_test_neighbors.libmesh_set_coupling_nef_polyhedron(Query_elem);
252 
253  // --- Set up variables needed by Gander's algorithm
254 
255  // First, determinate which patch will be the guide, and which will be
256  // the probed
257  Patch_construction * Patch_guide = &m_Patch_Constructor_A;
258  Patch_construction * Patch_probed = &m_Patch_Constructor_B;
259 
260  // Exchange them if the guide is smaller
261  bool bGuidedByB = Patch_guide->size() > Patch_probed->size();
262  if(bGuidedByB)
263  {
264  if(m_bPrintDebug)
265  {
266  std::cout << " DEBUG: Probe: A | Guide: B" << std::endl;
267  }
268  Patch_guide = &m_Patch_Constructor_B;
269  Patch_probed = &m_Patch_Constructor_A;
270  }
271  else
272  {
273  if(m_bPrintDebug)
274  {
275  std::cout << " DEBUG: Probe: B | Guide: A" << std::endl;
276  }
277  }
278 
279  // Ids of the working elements
280  unsigned int Guide_working_elem_id;
281  unsigned int Probed_working_elem_id;
282 
283  // Find the first pair, (guide, probed)
284  std::pair<unsigned int,unsigned int> First_intersection;
285  BruteForce_FindFirstPair(Patch_guide,Patch_probed,First_intersection);
286 
287  // Index and iterators guide neighbor elements to be tested
288  unsigned int guide_neighbor_index = 0;
289  std::unordered_set<unsigned int>::iterator it_neigh, it_neigh_end;
290 
291  // Boolean saying if two elements intersect (exactly)
292  bool bDoIntersect = false;
293 
294  // Boolean saying if a neighbor element intersects (exactly)
295  bool bDoNeighborIntersect = false;
296 
297  // Boolean used to short-circuit the neighbor search test if all
298  // the concerned elements already have neighbours.
299  bool bDoTestGuideNeighbours = true;
300 
301  // --- Preamble - initializations
302  // Insert the first elements in the queues
303  Patch_guide->FrontSearch_initialize();
304  Patch_probed->FrontSearch_initialize();
305 
306  Patch_guide->intersection_queue_push_back(First_intersection.first);
307  Patch_probed->intersection_queue_push_back(First_intersection.second);
308  Patch_guide->set_elem_as_inside_queue(First_intersection.first);
309 
310  // Debug vars
311  int nbOfGuideElemsTested = 0;
312  int nbOfTests = 0;
313  int nbOfPositiveTests = 0;
314 
315  bool bCreateNewNefForGuideNeigh = true;
316 
317  m_perf_log.pop("Preamble","Advancing front algorithm");
318 
319  while(!Patch_guide->intersection_queue_empty())
320  {
321  m_perf_log.push("Set up new guide element","Advancing front algorithm");
322 
323  // Pop out the first element from the guide intersection queue
324  Guide_working_elem_id = Patch_guide->intersection_queue_extract_front_elem();
325  Patch_guide->set_elem_as_treated(Guide_working_elem_id);
326 
327  // Get its pointer
328  const libMesh::Elem * elem_guide = Patch_guide->current_elem_pointer();
329 
330  ++nbOfGuideElemsTested;
331 
332  // Set up the neighbor test short-circuits
333  bDoTestGuideNeighbours = Patch_guide->set_neighbors_to_search_next_pairs();
334 
335  // Clear and initialize the probed patch lists
336  Patch_probed->FrontSearch_reset();
337 
338  // Pop out the first element from the probed intersection queue, and
339  // insert it inside the test queue
340  Patch_probed->FrontSearch_prepare_for_probed_test();
341 
342  m_perf_log.pop("Set up new guide element","Advancing front algorithm");
343 
344  while(!Patch_probed->test_queue_empty())
345  {
346  m_perf_log.push("Find intersection","Advancing front algorithm");
347  // Pop out the first elem to be tested
348  Probed_working_elem_id = Patch_probed->test_queue_extract_front_elem();
349  Patch_probed->set_elem_as_treated(Probed_working_elem_id);
350 
351  const libMesh::Elem * elem_probed = Patch_probed->current_elem_pointer();
352 
353  // Test the intersection
354  bDoIntersect = m_Intersection_test.libMesh_exact_do_intersect(elem_guide,elem_probed);
355  ++nbOfTests;
356  m_perf_log.pop("Find intersection","Advancing front algorithm");
357 
358  // If they do intersect, we have to ...
359  if(bDoIntersect)
360  {
361  // 1) add elements to intersection multimap, watching out
362  // for the element order
363  if(m_bSaveInterData)
364  {
365  if(bGuidedByB)
366  {
367  m_Intersection_Pairs_multimap.insert(std::pair<unsigned int, unsigned int>(Probed_working_elem_id,Guide_working_elem_id));
368  }
369  else
370  {
371  m_Intersection_Pairs_multimap.insert(std::pair<unsigned int, unsigned int>(Guide_working_elem_id,Probed_working_elem_id));
372  }
373  }
374  ++nbOfPositiveTests;
375 
376  m_perf_log.push("Update test queue","Advancing front algorithm");
377  // 2) add elem_probed's neighbors, if not treated yet
378  Patch_probed->add_neighbors_to_test_list();
379  m_perf_log.pop("Update test queue","Advancing front algorithm");
380 
381  m_perf_log.push("Update intersection queue","Advancing front algorithm");
382  // 3) determinate if any of the guide neighbors are good
383  // candidates with the probed element.
384  if(bDoTestGuideNeighbours)
385  {
386  it_neigh = Patch_guide->neighbors_to_search_next_pair().begin();
387  it_neigh_end = Patch_guide->neighbors_to_search_next_pair().end();
388 
389  bCreateNewNefForGuideNeigh = true;
390 
391  for( ; it_neigh != it_neigh_end; ++it_neigh )
392  {
393  // This element doesn't have an intersecting pair
394  // yet. Test it, but only save it if it is inside
395  // the coupling region
396  guide_neighbor_index = *it_neigh;
397  const libMesh::Elem * elem_neigh = Patch_guide->elem(guide_neighbor_index);
398  bDoNeighborIntersect = m_Intersection_test_neighbors.libMesh_exact_do_intersect_inside_coupling(elem_probed,elem_neigh,bCreateNewNefForGuideNeigh);
399 
400  if(bDoNeighborIntersect)
401  {
402  bCreateNewNefForGuideNeigh = false;
403  }
404 
405  if(bDoNeighborIntersect)
406  {
407  // Now it has an intersecting pair!
408  Patch_guide->intersection_queue_push_back(guide_neighbor_index);
409  Patch_probed->intersection_queue_push_back(Probed_working_elem_id);
410 
411  // Set the guide elements as "already inside the queue"
412  Patch_guide->set_elem_as_inside_queue(guide_neighbor_index);
413  }
414  }
415 
416  bDoTestGuideNeighbours = Patch_guide->set_neighbors_to_search_next_pairs();
417  }
418  m_perf_log.pop("Update intersection queue","Advancing front algorithm");
419  }
420  }
421  }
422 
423  // Partitioning vars
424  unsigned int query_idx = Query_elem->id();
425  m_Nb_Of_Intersections_Elem_C[query_idx] = nbOfPositiveTests;
426 
427  if(m_bPrintDebug)
428  {
429  std::cout << " DEBUG: advancing front search results" << std::endl;
430  std::cout << " -> Guide elements tested / all : " << nbOfGuideElemsTested << " / " << Patch_guide->size() << std::endl;
431  std::cout << " -> Positives / tests : " << nbOfPositiveTests << " / " << nbOfTests
432  << " (" << 100.*nbOfPositiveTests/nbOfTests << "%)" << std::endl << std::endl;
433  }
434  }
